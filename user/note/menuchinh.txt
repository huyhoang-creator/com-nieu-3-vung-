            <?php
            $result = $conn->query("SELECT dm.ten AS danh_muc, ma.ten, ma.mo_ta, ma.gia, ma.hinh_anh , ma.trang_thai
                                FROM MON_AN ma 
                                JOIN DANH_MUC dm ON ma.id_danh_muc = dm.id 
                                WHERE ma.trang_thai = 'con_mon'");

            $categories = [];
            while ($row = $result->fetch_assoc()) {
                $categories[$row['danh_muc']][] = $row;
            }

            foreach ($categories as $cat_name => $items) {
                echo "<div class='category-section' data-region='1'>";
                echo "<h2 class='category-title'>" . htmlspecialchars($cat_name) . "</h2>";
                echo "<div class='menu-grid'>";
                foreach ($items as $item) {
                    echo "<div class='menu-item' data-name='" . htmlspecialchars($item['ten']) . "' data-category='" . htmlspecialchars(strtolower(str_replace(' ', '_', $cat_name))) . "' data-price='" . ($item['gia'] ? $item['gia'] : 0) . "'>";
                    echo "<div class='item-image'>";
                    if ($item['hinh_anh']) {
                        echo "<img src='img/" . htmlspecialchars($item['hinh_anh']) . "' alt='" . htmlspecialchars($item['ten']) . "'>";
                    } else {
                        echo "<img src='https://via.placeholder.com/400x300?text=Không+có+hình' alt='Không có hình'>";
                    }
                    if ($cat_name === 'Đặc sản') {
                        echo "<div class='specialty-badge'><i class='fas fa-star'></i> Đặc trưng</div>";
                    }
                    echo "</div>";
                    echo "<div class='item-content'>";
                    echo "<h3 class='item-name'>" . htmlspecialchars($item['ten']) . "</h3>";
                    echo "<p class='item-description'>" . htmlspecialchars($item['mo_ta']) . "</p>";
                    echo "<p class='item-description'>" . htmlspecialchars($item['trang_thai']) . "</p>";
                    echo "<div class='item-footer'>";
                    if ($item['gia']) {
                        echo "<span class='item-price'>" . number_format($item['gia'], 0, ',', '.') . "đ</span>";
                    } else {
                        echo "<span class='item-price'>Giờ chuẩn bị</span>";
                    }
                    echo "<button class='order-btn'>Đặt món</button>";
                    echo "</div>";
                    echo "</div>";
                    echo "</div>";
                }
                // Điền các ô trống nếu số lượng item ít hơn 4
                while (count($items) < 4) {
                    echo "<div class='menu-item'></div>";
                    $items[] = null;
                }
                echo "</div></div>";
            }
            ?>
